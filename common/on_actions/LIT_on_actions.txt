on_actions = {
    on_puppet = {
            effect = {
                if = {
                    limit = { 
                            original_tag = LIT
                            is_subject_of = POL
                    }
					
					set_cosmetic_tag = LIT_POL
				    
            }
        }
    }
    on_subject_free = {
            effect = {
                if = {
                    limit = { 
                            original_tag = LIT
                    }
                
				drop_cosmetic_tag = LIT_POL
				
            }
        }
    }

 }
